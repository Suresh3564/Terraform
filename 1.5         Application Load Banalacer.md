# 🧠 1. What is Application Load Balancer (ALB)?

Application Load Balancer (ALB) is a **Layer 7 (HTTP/HTTPS) load balancer** in AWS.

👉 It:

* Distributes traffic across multiple EC2 instances
* Works based on HTTP rules (path, host, headers)
* Provides high availability and fault tolerance

# ⚙️ 4. Complete Terraform Code (One-Shot)

```hcl id="albfull"
provider "aws" {
  region = "ap-south-1"
}

################ VPC ################
resource "aws_vpc" "main" {
  cidr_block = "10.0.0.0/16"
}

################ SUBNETS ################
resource "aws_subnet" "public1" {
  vpc_id                  = aws_vpc.main.id
  cidr_block              = "10.0.1.0/24"
  availability_zone       = "ap-south-1a"
  map_public_ip_on_launch = true
}

resource "aws_subnet" "public2" {
  vpc_id                  = aws_vpc.main.id
  cidr_block              = "10.0.2.0/24"
  availability_zone       = "ap-south-1b"
  map_public_ip_on_launch = true
}

################ IGW ################
resource "aws_internet_gateway" "igw" {
  vpc_id = aws_vpc.main.id
}

################ ROUTE ################
resource "aws_route_table" "rt" {
  vpc_id = aws_vpc.main.id

  route {
    cidr_block = "0.0.0.0/0"
    gateway_id = aws_internet_gateway.igw.id
  }
}

resource "aws_route_table_association" "a1" {
  subnet_id      = aws_subnet.public1.id
  route_table_id = aws_route_table.rt.id
}

resource "aws_route_table_association" "a2" {
  subnet_id      = aws_subnet.public2.id
  route_table_id = aws_route_table.rt.id
}

################ SECURITY GROUPS ################
resource "aws_security_group" "ec2_sg" {
  vpc_id = aws_vpc.main.id

  ingress {
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  ingress {
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

resource "aws_security_group" "alb_sg" {
  vpc_id = aws_vpc.main.id

  ingress {
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

################ AMI ################
data "aws_ami" "amazon_linux" {
  most_recent = true
  owners      = ["amazon"]

  filter {
    name   = "name"
    values = ["amzn2-ami-hvm-*-x86_64-gp2"]
  }
}

################ EC2 - GREEN ################
resource "aws_instance" "web1" {
  ami           = data.aws_ami.amazon_linux.id
  instance_type = "t3.micro"
  subnet_id     = aws_subnet.public1.id

  vpc_security_group_ids = [aws_security_group.ec2_sg.id]

  user_data = <<-EOF
#!/bin/bash
yum install -y httpd
systemctl start httpd
echo "<h1 style='color:green'>GREEN SERVER</h1>" > /var/www/html/index.html
EOF
}

################ EC2 - BLUE ################
resource "aws_instance" "web2" {
  ami           = data.aws_ami.amazon_linux.id
  instance_type = "t3.micro"
  subnet_id     = aws_subnet.public2.id

  vpc_security_group_ids = [aws_security_group.ec2_sg.id]

  user_data = <<-EOF
#!/bin/bash
yum install -y httpd
systemctl start httpd
echo "<h1 style='color:blue'>BLUE SERVER</h1>" > /var/www/html/index.html
EOF
}

################ TARGET GROUP ################
resource "aws_lb_target_group" "tg" {
  name     = "suresh-tg"
  port     = 80
  protocol = "HTTP"
  vpc_id   = aws_vpc.main.id
}

################ ATTACH INSTANCES ################
resource "aws_lb_target_group_attachment" "t1" {
  target_group_arn = aws_lb_target_group.tg.arn
  target_id        = aws_instance.web1.id
  port             = 80
}

resource "aws_lb_target_group_attachment" "t2" {
  target_group_arn = aws_lb_target_group.tg.arn
  target_id        = aws_instance.web2.id
  port             = 80
}

################ ALB ################
resource "aws_lb" "alb" {
  name               = "suresh-alb"
  load_balancer_type = "application"
  subnets            = [aws_subnet.public1.id, aws_subnet.public2.id]
  security_groups    = [aws_security_group.alb_sg.id]
}

################ LISTENER ################
resource "aws_lb_listener" "listener" {
  load_balancer_arn = aws_lb.alb.arn
  port              = 80
  protocol          = "HTTP"

  default_action {
    type             = "forward"
    target_group_arn = aws_lb_target_group.tg.arn
  }
}

################ OUTPUT ################
output "alb_dns" {
  value = aws_lb.alb.dns_name
}
```

# 🔥 7. How Traffic Works

1. User hits ALB DNS
2. Listener receives request
3. Forwards to Target Group
4. Target Group selects EC2
5. Response sent back

---

# 🎯 8. Load Balancing Behavior

👉 Default algorithm: **Round Robin**

* Request 1 → GREEN
* Request 2 → BLUE
* Request 3 → GREEN

---

# 🔒 9. Security Flow

* ALB Security Group → allows HTTP (80)
* EC2 Security Group → allows HTTP + SSH

💡 Best Practice:
Allow traffic only from ALB SG (not 0.0.0.0/0)
